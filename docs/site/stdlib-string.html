std.string

String manipulation functions.

import std.string

Core

str_length

def str_length(s: String) -> Int

Returns the number of characters in s.

str_substring

def str_substring(s: String, start: Int, length: Int) -> String

Returns a substring starting at start with the given length.

Trimming

str_trim

def str_trim(s: String) -> String

Removes leading and trailing whitespace from s.

str_trim_start

def str_trim_start(s: String) -> String

Removes leading whitespace from s.

str_trim_end

def str_trim_end(s: String) -> String

Removes trailing whitespace from s.

import std.string

text = "  hello world  "
println(str_trim(text))        // "hello world"
println(str_trim_start(text))  // "hello world  "
println(str_trim_end(text))    // "  hello world"

Searching

starts_with

def starts_with(s: String, prefix: String) -> Bool

Returns true if s begins with prefix.

ends_with

def ends_with(s: String, suffix: String) -> Bool

Returns true if s ends with suffix.

contains

def contains(s: String, needle: String) -> Bool

Returns true if s contains needle as a substring.

index_of

def index_of(s: String, needle: String) -> Int

Returns the index of the first occurrence of needle in s, or -1 if not found.

import std.string

path = "src/main.nex"
println(starts_with(path, "src"))   // true
println(ends_with(path, ".nex"))     // true
println(contains(path, "main"))     // true
println(index_of(path, "main"))     // 4
println(index_of(path, "test"))     // -1

Replacement

str_replace

def str_replace(s: String, old: String, new: String) -> String

Returns a new string with all occurrences of old replaced by new.

import std.string

msg = "Hello World"
println(str_replace(msg, "World", "Nex"))  // "Hello Nex"

Case Conversion

to_upper

def to_upper(s: String) -> String

Returns s with all characters converted to uppercase.

to_lower

def to_lower(s: String) -> String

Returns s with all characters converted to lowercase.

import std.string

println(to_upper("hello"))  // "HELLO"
println(to_lower("WORLD"))  // "world"

Repetition

str_repeat

def str_repeat(s: String, count: Int) -> String

Returns s repeated count times.

import std.string

println(str_repeat("=-", 20))
// "=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-"

println(str_repeat("abc", 3))
// "abcabcabc"

Character Access

char_at

def char_at(s: String, index: Int) -> Int

Returns the character code at the given index, or -1 if the index is out of bounds.

import std.string

code = char_at("Hello", 0)
println(code)               // 72 (ASCII for 'H')

invalid = char_at("Hi", 99)
println(invalid)            // -1

Reverse

str_reverse

def str_reverse(s: String) -> String

Returns a new string with the characters of s in reverse order.

import std.string

println(str_reverse("Nex"))  // "xeN"
println(str_reverse("racecar"))  // "racecar" (palindrome)

Splitting

str_split

def str_split(s: String, delimiter: String) -> String

Splits s by delimiter and returns a single string with results separated by newlines. Use with line-based processing.

import std.string

csv = "apple,banana,cherry"
parts = str_split(csv, ",")
println(parts)
// apple
// banana
// cherry

Complete Example

import std.string

// Clean and validate user input
def normalize_email(input: String) -> String {
    trimmed = str_trim(input)
    return to_lower(trimmed)
}

def is_valid_email(email: String) -> Bool {
    return contains(email, "@")
}

raw = "  User@Example.COM  "
email = normalize_email(raw)
println(email)                    // "user@example.com"
println(is_valid_email(email))    // true

// Parse a PATH-like variable
path_var = "/usr/bin:/usr/local/bin:/home/user/bin"
dirs = str_split(path_var, ":")
println(dirs)
// /usr/bin
// /usr/local/bin
// /home/user/bin
