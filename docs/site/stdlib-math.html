std.math

Mathematical functions for integers and floating-point numbers.

import std.math

Absolute Value

abs_int

def abs_int(v: Int) -> Int

Returns the absolute value of an integer.

abs_float

def abs_float(v: Double) -> Double

Returns the absolute value of a floating-point number.

import std.math

println(abs_int(-42))      // 42
println(abs_float(-3.14))  // 3.14

Min / Max

min_int

def min_int(a: Int, b: Int) -> Int

Returns the smaller of two integers.

max_int

def max_int(a: Int, b: Int) -> Int

Returns the larger of two integers.

min_float

def min_float(a: Double, b: Double) -> Double

Returns the smaller of two doubles.

max_float

def max_float(a: Double, b: Double) -> Double

Returns the larger of two doubles.

import std.math

println(min_int(10, 3))           // 3
println(max_int(10, 3))           // 10
println(min_float(1.5, 2.7))    // 1.5
println(max_float(1.5, 2.7))    // 2.7

Clamping

clamp_int

def clamp_int(v: Int, lo: Int, hi: Int) -> Int

Clamps an integer to the range [lo, hi].

clamp_float

def clamp_float(v: Double, lo: Double, hi: Double) -> Double

Clamps a double to the range [lo, hi].

import std.math

println(clamp_int(15, 0, 10))          // 10
println(clamp_int(-5, 0, 10))          // 0
println(clamp_float(3.7, 0.0, 1.0))   // 1.0

Rounding

floor

def floor(v: Double) -> Double

Returns the largest integer value less than or equal to v.

ceil

def ceil(v: Double) -> Double

Returns the smallest integer value greater than or equal to v.

round

def round(v: Double) -> Double

Returns the nearest integer value, rounding half away from zero.

import std.math

println(floor(3.7))    // 3.0
println(ceil(3.2))     // 4.0
println(round(3.5))    // 4.0
println(round(3.4))    // 3.0

Powers & Roots

sqrt

def sqrt(v: Double) -> Double

Returns the square root of v.

pow

def pow(base: Double, exp: Double) -> Double

Returns base raised to the power exp.

import std.math

println(sqrt(16.0))        // 4.0
println(pow(2.0, 10.0))   // 1024.0
println(pow(3.0, 0.5))    // 1.7320508... (sqrt of 3)

cos

import std.math

Logarithms & Exponentials

log

def log(v: Double) -> Double

Returns the natural logarithm (base e) of v.

log2

def log2(v: Double) -> Double

Returns the base-2 logarithm of v.

log10

def log10(v: Double) -> Double

Returns the base-10 logarithm of v.

exp

def exp(v: Double) -> Double

Returns e raised to the power v.

import std.math

println(log(1.0))       // 0.0
println(log2(8.0))      // 3.0
println(log10(100.0))   // 2.0
println(exp(1.0))       // 2.71828...

Random Numbers

random

def random() -> Double

Returns a pseudo-random double in the range [0.0, 1.0).

random_range

def random_range(lo: Int, hi: Int) -> Int

Returns a pseudo-random integer in the range [lo, hi).

import std.math

// Random double between 0.0 and 1.0
r = random()
println(r)               // e.g. 0.73291...

// Random integer: simulating a dice roll (1-6)
die = random_range(1, 7)
println(die)             // e.g. 4

Complete Example

import std.math

// Distance between two points
def distance(x1: Double, y1: Double, x2: Double, y2: Double) -> Double {
    dx = x2 - x1
    dy = y2 - y1
    return sqrt(dx * dx + dy * dy)
}

println(distance(0.0, 0.0, 3.0, 4.0))  // 5.0

// Normalize a value to 0-100 range
raw = 127.6
clamped = clamp_float(raw, 0.0, 100.0)
println(clamped)                          // 100.0
