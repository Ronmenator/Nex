std.testing

Assertion functions for unit testing.

import std.testing

General Assertions

assert

def assert(condition: Bool, msg: String) -> Unit

Asserts that condition is true. If the condition is false, the program terminates with msg as the error message.

assert_true

def assert_true(condition: Bool, msg: String) -> Unit

Alias for assert. Asserts that condition evaluates to true.

Equality Assertions

assert_eq_int

def assert_eq_int(a: Int, b: Int, msg: String) -> Unit

Asserts that two integers are equal. If a != b, the program terminates with msg.

assert_eq_str

def assert_eq_str(a: String, b: String, msg: String) -> Unit

Asserts that two strings are equal. Comparison is case-sensitive.

assert_eq_float

def assert_eq_float(a: Double, b: Double, msg: String) -> Unit

Asserts that two floating-point values are equal. Due to floating-point precision, consider rounding or using a tolerance check for computed values.

assert_eq_bool

def assert_eq_bool(a: Bool, b: Bool, msg: String) -> Unit

Asserts that two boolean values are equal.

Inequality Assertions

assert_ne_int

def assert_ne_int(a: Int, b: Int, msg: String) -> Unit

Asserts that two integers are not equal. If a == b, the program terminates with msg.

assert_ne_str

def assert_ne_str(a: String, b: String, msg: String) -> Unit

Asserts that two strings are not equal.

Complete Example

import std.testing
import std.math
import std.string

// Test basic arithmetic
def test_arithmetic() -> Unit {
    assert_eq_int(2 + 3, 5, "addition failed")
    assert_eq_int(10 - 4, 6, "subtraction failed")
    assert_eq_int(3 * 7, 21, "multiplication failed")
    assert_ne_int(10 / 3, 4, "integer division should truncate")
    println("arithmetic tests passed")
}

// Test string operations
def test_strings() -> Unit {
    assert_eq_str(to_upper("hello"), "HELLO", "to_upper failed")
    assert_eq_str(to_lower("WORLD"), "world", "to_lower failed")
    assert_ne_str("foo", "bar", "different strings should not match")
    println("string tests passed")
}

// Test math functions
def test_math() -> Unit {
    assert_eq_float(sqrt(4.0), 2.0, "sqrt(4) should be 2")
    assert_eq_int(abs_int(-42), 42, "abs(-42) should be 42")
    assert_eq_bool(3 < 5, true, "3 should be less than 5")
    println("math tests passed")
}

// Run all tests
test_arithmetic()
test_strings()
test_math()
println("All tests passed!")
