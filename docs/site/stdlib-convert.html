std.convert

Type conversion and parsing functions.

import std.convert

String to Number

parse_int

def parse_int(s: String) -> Int

Parses a string as a 32-bit integer. The string must contain a valid integer literal (optionally prefixed with - for negative values). Returns 0 if parsing fails.

import std.convert

n = parse_int("42")
println(n)              // 42

neg = parse_int("-100")
println(neg)            // -100

bad = parse_int("hello")
println(bad)            // 0

parse_float

def parse_float(s: String) -> Double

Parses a string as a 64-bit floating-point number. Returns 0.0 if parsing fails.

import std.convert

pi = parse_float("3.14159")
println(pi)             // 3.14159

sci = parse_float("1.5e3")
println(sci)            // 1500.0

String to Boolean

parse_bool

def parse_bool(s: String) -> Bool

Parses a string as a boolean. The following strings are treated as true (case-insensitive): "true", "1", "yes". All other values return false.

import std.convert

println(parse_bool("true"))    // true
println(parse_bool("1"))       // true
println(parse_bool("yes"))     // true
println(parse_bool("false"))   // false
println(parse_bool("0"))       // false
println(parse_bool("no"))      // false

Character Conversion

char_to_str

def char_to_str(c: Int) -> String

Converts a character code (integer) to a single-character string.

import std.convert

letter = char_to_str(65)
println(letter)          // "A"

newline = char_to_str(10)
print("Line 1")
print(newline)
print("Line 2")

str_to_chars

def str_to_chars(s: String) -> Var

Converts a string into an array of character codes (as a Var). Each element is the integer code of the corresponding character.

import std.convert

var codes = str_to_chars("Hi")
// codes contains [72, 105]
// 72 = 'H', 105 = 'i'

import std.convert

// Build a string from character codes
def build_alphabet(start: Int, count: Int) -> String {
    result = ""
    i = 0
    while (i < count) {
        result = result + char_to_str(start + i)
        i = i + 1
    }
    return result
}

println(build_alphabet(65, 26))
// "ABCDEFGHIJKLMNOPQRSTUVWXYZ"

println(build_alphabet(97, 26))
// "abcdefghijklmnopqrstuvwxyz"
