std.path

File path manipulation utilities.

import std.path

Joining & Combining

path_join

def path_join(a: String, b: String) -> String

Joins two path segments with the platform-appropriate separator. If b is an absolute path, it replaces a entirely.

import std.path

full = path_join("/home/user", "documents")
println(full)   // /home/user/documents

nested = path_join(path_join("src", "models"), "user.nex")
println(nested)   // src/models/user.nex

Path Components

path_parent

def path_parent(p: String) -> String

Returns the parent directory of the given path. Returns an empty string if the path has no parent.

import std.path

println(path_parent("/home/user/file.txt"))   // /home/user
println(path_parent("src/main.nex"))          // src

path_file_name

def path_file_name(p: String) -> String

Returns the final component of the path (the file or directory name).

import std.path

println(path_file_name("/home/user/report.pdf"))   // report.pdf
println(path_file_name("src/main.nex"))             // main.nex

path_extension

def path_extension(p: String) -> String

Returns the file extension (without the leading dot). Returns an empty string if there is no extension.

import std.path

println(path_extension("archive.tar.gz"))   // gz
println(path_extension("Makefile"))         // (empty string)

path_stem

def path_stem(p: String) -> String

Returns the file name without its extension.

import std.path

println(path_stem("report.pdf"))   // report
println(path_stem("data.tar.gz"))  // data.tar

Path Properties

path_is_absolute

def path_is_absolute(p: String) -> Bool

Returns true if the path is absolute (begins with / on Unix or a drive letter on Windows).

import std.path

println(path_is_absolute("/usr/bin"))      // true
println(path_is_absolute("src/main.nex"))   // false

path_normalize

def path_normalize(p: String) -> String

Normalizes the path by resolving . and .. segments and removing redundant separators.

import std.path

println(path_normalize("src/../src/./main.nex"))   // src/main.nex
println(path_normalize("/home//user///file"))    // /home/user/file

path_separator

def path_separator() -> String

Returns the platform-specific path separator: / on Unix, \ on Windows.

import std.path

sep = path_separator()
println(sep)   // "/" or "\"

Complete Example

import std.path
import std.io

// Build an output path from components
def make_output_path(dir: String, name: String, ext: String) -> String {
    filename = name + "." + ext
    return path_join(dir, filename)
}

// Analyze a file path
def describe_path(p: String) -> Unit {
    println("Path analysis:")

    print("  Full path:  ")
    println(p)

    print("  Parent:     ")
    println(path_parent(p))

    print("  File name:  ")
    println(path_file_name(p))

    print("  Stem:       ")
    println(path_stem(p))

    print("  Extension:  ")
    println(path_extension(p))

    print("  Absolute?   ")
    println(path_is_absolute(p))
}

src = "./project/../project/src/main.nex"
describe_path(path_normalize(src))

output = make_output_path("build/release", "app", "exe")
print("Output: ")
println(output)   // build/release/app.exe
